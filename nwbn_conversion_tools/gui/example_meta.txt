This is an example meta-data file to use for NWB files.

# Path to files
fpath=path/to/file

# File name
fname=my_experiment.nwb

# The identifier is a unique ID for this file, it's optional
# (generate using UUID or come up a code thats unique, i.e. MY_LAB_234)
identifier=ADDME

# internal code for "one day of recording"
session_id=ADDME
session_description=ADDME

# experimenter and lab
experimenter=ADDME
lab=ADDME
institution=ADDME

# notes
notes=ADDME
stimulus_notes=ADDME
data_collection=ADDME
experiment_description=ADDME
protocol=ADDME
keywords=KW1,KW2,KW3

# Other arguments
option_1=False
option_2=True
